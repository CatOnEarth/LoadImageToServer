64 private static String getFileName(Context context, Uri uri) {
65 String result = null;
66 if (uri.getScheme().equals("content")) {
67 try (Cursor cursor = context.getContentResolver().query(uri, null, null, null, null)) {
68 if (cursor != null && cursor.moveToFirst()) {
69 result = cursor.getString(cursor.getColumnIndex(OpenableColumns.DISPLAY_NAME));
70 }
71 } catch (Exception e) {
72 e.printStackTrace();
73 }
74 }
75 if (result == null) {
76 result = uri.getPath();
77 int cut = result.lastIndexOf(File.separator);
78 if (cut != -1) {
79 result = result.substring(cut + 1);

81 }

98 private static long copy(InputStream input, OutputStream output) throws IOException {
